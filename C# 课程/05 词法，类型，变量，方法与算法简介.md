```csharp
    {

        {

            return result;
        }
        public string Today()//没有数据输入，有输出
        {
            int day = DateTime.Now.Day;
            return day.ToString();
        }
        public void PrintSum(int x, int y)//有数据输入，没有输出
        {
            int result = x + y;
            Console.WriteLine(result);
        }
    }
}

```

# 算法简介  
### 循环初体验  
```csharp
using System;

namespace ConsoleApp2
{
    class Program
    {
        static void Main(string[] args)
        {
            Calculator c = new Calculator();
            c.PrintXTo1(4);
        }
    }

    class Calculator
    {
       public void PrintXTo1(int x)
        {
            for(int i = x; i > 0; i--)
            {
                Console.WriteLine(i);
            }
        }
    }
}
```

### 计算1到100的和  
```csharp
class Calculator
    {
        public int SumFrom1ToX(int x)
        {
            int result = 0;
            for(int i = 1; i < x + 1; i++)
            {
                result += i;
            }
            return result;
        }
        public int SumFrom1ToX2(int x)
        {
            if (x == 1)
            {
                return 1;
            }
            else
            {
                int result = x + SumFrom1ToX2(x - 1);
                return result;
            }
        }
    }
```

# 作业 汉诺塔 
第一根圆柱上64个，多少次可以移到另一根上面  
```csharp
using System;

namespace ConsoleApp2
{
    class Program
    {
        static void Main(string[] args)
        {
            Hanoi c = new Hanoi();
            int x=c.MoveHanoi(4);
            Console.WriteLine(x);
        }
    }

    class Hanoi
    {
        public int MoveHanoi(int x)
        {
            if (x == 1)
            {
                return 1;
            }
            else
            {
                return 2*MoveHanoi(x - 1) + 1;
            }
        }
    }
}
```

### 类型，变量，值 三角关系  
T ,  Var ,  Value     
* 类型和变量
  * 类型可以申明变量，可以决定变量在内存中的大小   
  * C#所有变量都是有类型的，确定之后不能改变   
* 类型和值  
  * 所有值都是有类型的  
  * 类型决定了值的行为  
* 值和变量

  * 引用类型存储的是值的地址  
* 隐藏

###  隐式类型转换
```csharp
namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            Console.WriteLine((100).GetType().Name);//Int32
            Console.WriteLine((100).GetType().FullName);//System.Int32
            long x = 100;//有隐式类型转换
            long y = 100l;//无转换

            short s = 1;
            //s = s + 1;//类型提升，错误
            s += 1;
            s = (short)(s + 1);
            Console.WriteLine(s);
            
            Console.WriteLine((new student()).GetType().FullName);//ConsoleApp1.student
            Console.WriteLine((new car()).GetType().FullName);//ConsoleApp1.car
            string ss = "12";
            Console.WriteLine(ss.GetType().FullName);//System.String  
        }
    }
    class student
    {
        public int id;
    }

    struct car
    {
        int id;
    }
}
```
